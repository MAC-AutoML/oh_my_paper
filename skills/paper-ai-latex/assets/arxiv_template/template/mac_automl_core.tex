\newcommand{\mymaketitle}{%
  \tcbset{enhanced}
  \tcbset{left=8mm}
  \tcbset{right=8mm}
  \tcbset{top=4.0mm}
  \tcbset{bottom=3.8mm}
  \tcbset{colback=white}
  \tcbset{colframe=omniframe}
  \tcbset{boxrule=3pt}
  \tcbset{before skip=0pt}
  \tcbset{grow to left by=1.5pt}
  \tcbset{grow to right by=1.5pt}
  \tcbset{arc=2.2mm}
      {
      \setlength{\parskip}{0mm}
      \hfuzz=20pt
      \hbadness=10000
      \centering
      \nohyphens

      {
      \vspace*{-4.8mm}
      \@toptitlebar

        \ifmacsubmissionmode
          {\sffamily\fontsize{9.2}{11}\selectfont\bfseries\color{omniaccent!72!black}\macreviewheaderlabel\par}
          \vskip 2.4mm
        \fi
        \titlelist\par
         \@bottomtitlebar
      }
      \vskip 4.0mm
      \authorlist
      \vskip 2.4mm

      \affiliationlist\par
      \ifdefempty{\emaillist}{}{\vskip 3mm \emaillist\par}

      \contributionlist\par
    }
   \vskip 2.6mm

  \begin{tcolorbox}[
    enhanced jigsaw,
    colback=abstractboxfill,
    colframe=omniaccent!99,
    boxrule=1.6pt,
    arc=2.5mm,
    left=8mm,
    right=8mm,
    top=4.1mm,
    bottom=4.2mm,
    interior style={fill=abstractboxfill},
    drop shadow={black!12!white},
    before skip=0pt,
    after skip=0pt
  ]
    \setlength{\parindent}{0cm}
    \setlength{\parskip}{0cm}

    \begin{center}
      \beginabstract
      \vskip 1.8mm
    \end{center}

    \abstractlist\par
    \vskip 0.6mm
    {
      \setlength{\parskip}{0cm}
      \ifdefempty{\checkdatalist}{\vspace*{0.02cm}}{\checkdatalist\par}
      \ifmacsubmissionmode
        {\small {\checkdatafont\sffamily \bfseries Mode:} \macreviewheaderlabel\par}
      \fi
    }
  \end{tcolorbox}
  \tcbset{reset}
  \FloatBarrier

  \thispagestyle{firststyle}
}
